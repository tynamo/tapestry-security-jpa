18  @Deprecated
19  public class SecureWriteAdvice implements MethodAdvice {
20  	private final SecurityService securityService;
21  	private final HttpServletRequest request;
22  	private final PropertyAccess propertyAccess;
23  	private final Operation writeOperation;
24  
25  	public SecureWriteAdvice(final Operation writeOperation, final SecurityService securityService,
26  		final HttpServletRequest request, final PropertyAccess propertyAccess) {
27  		if (!Operation.INSERT.equals(writeOperation) && !Operation.UPDATE.equals(writeOperation)
28  			&& !Operation.DELETE.equals(writeOperation))
29  			throw new IllegalArgumentException("The targeted writeOperation must be one of INSERT, UPDATE, DELETE but was "
30  				+ writeOperation);
31  		this.writeOperation = writeOperation;
32  		this.securityService = securityService;
33  		this.request = request;
34  		this.propertyAccess = propertyAccess;
35  	}
36  
37  	@SuppressWarnings({ "unchecked", "rawtypes" })
38  	@Override
39  	public void advise(MethodInvocation invocation) {
40  		Object entity = invocation.getParameter(0);
41  		String requiredRoleValue = RequiresAnnotationUtil.getRequiredRole(entity.getClass(), writeOperation);
42  
43  		if (requiredRoleValue != null && request.isUserInRole(requiredRoleValue)) {
44  			invocation.proceed();
45  			return;
46  		}
47  
48  		String requiredAssociationValue = RequiresAnnotationUtil.getRequiredAssociation(entity.getClass(), writeOperation);
49  
50  		if (requiredAssociationValue == null) {
51  			// proceed as normal if there's neither RequiresRole nor RequiresAssociation, throw an exception if role didn't match
52  			if (requiredRoleValue == null) invocation.proceed();
53  			else throw new EntitySecurityException("Currently executing subject is not permitted to " + writeOperation
54  				+ " entities of type " + entity.getClass().getSimpleName());
55  		}
56  		EntityManager entityManager = (EntityManager) invocation.getInstance();
57  
58  		// FIXME handle empty value, i.e. association to "self"
59  		Object associatedObject = propertyAccess.get(entity, requiredAssociationValue);
60  
61  		Metamodel metamodel = entityManager.getMetamodel();
62  		EntityType entityType = metamodel.entity(entity.getClass());
63  		Attribute association = entityType.getAttribute(requiredAssociationValue);
64  		entityType = metamodel.entity(association.getJavaType());
65  		Type idType = entityType.getIdType();
66  		SingularAttribute idAttr = entityType.getId(idType.getJavaType());
67  		if (!propertyAccess.get(associatedObject, idAttr.getName()).equals(
68  			securityService.getSubject().getPrincipals().getPrimaryPrincipal())) { throw new EntitySecurityException(
69  			"Currently executing subject is not permitted to " + writeOperation + " entities of type "
70  				+ entity.getClass().getSimpleName() + " because the required association didn't exist"); }
71  		invocation.proceed();
72  	}
73  }

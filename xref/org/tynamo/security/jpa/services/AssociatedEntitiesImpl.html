17  public class AssociatedEntitiesImpl implements AssociatedEntities {
18  	private final EntityManager delegate;
19  	private final SecurityService securityService;
20  	private final HttpServletRequest request;
21  	private String realmName;
22  	private Class<?> principalType;
23  
24  	public AssociatedEntitiesImpl(
25  		final SecurityService securityService,
26  		final HttpServletRequest request, final EntityManager delegate,
27  		@Inject @Symbol(JpaSecuritySymbols.ASSOCIATED_REALM) String realmName,
28  		@Inject @Symbol(JpaSecuritySymbols.ASSOCIATED_PRINCIPALTYPE) String principalType) throws ClassNotFoundException {
29  		this.securityService = securityService;
30  		this.request = request;
31  		this.delegate = delegate;
32  		this.realmName = realmName;
33  		this.principalType = principalType.isEmpty() ? null : Class.forName(principalType);
34  	}
35  
36  	@Override
37  	public <T> List<T> findAll(EntityManager em, Class<T> entityClass) throws Exception {
38  		String requiredAssociationValue = RequiresAnnotationUtil.getRequiredAssociation(entityClass, Operation.READ);
39  
40  		// return SecureFinder.find(delegate, request, SecureFinder.getConfiguredPrincipal(securityService,
41  		// realmName,
42  		// principalType),
43  		// requiredAssociationValue, null, entityClass,
44  		// null, null, null);
45  		
46  		Object principal = SecureFinder.getConfiguredPrincipal(securityService, realmName, principalType);
47  		// invoke with security disabled so we don't go to a recursion with em.find()
48  		return securityService.invokeWithSecurityDisabled(new Callable<List<T>>() {
49  			@Override
50  			public List<T> call() {
51  				return SecureFinder.find(delegate, request, principal,
52  					requiredAssociationValue,
53  					null, entityClass, null, null, null);
54  			}
55  		});
56  		
57  	}

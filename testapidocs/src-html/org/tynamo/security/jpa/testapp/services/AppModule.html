019package org.tynamo.security.jpa.testapp.services;
020
021import org.apache.shiro.realm.Realm;
022import org.apache.tapestry5.SymbolConstants;
023import org.apache.tapestry5.ioc.Configuration;
024import org.apache.tapestry5.ioc.MappedConfiguration;
025import org.apache.tapestry5.ioc.OrderedConfiguration;
026import org.apache.tapestry5.ioc.ServiceBinder;
027import org.apache.tapestry5.ioc.annotations.SubModule;
028import org.tynamo.security.jpa.EntitySecurityException;
029import org.tynamo.security.jpa.JpaSecurityModule;
030import org.tynamo.security.jpa.testapp.entities.AdminOnly;
031import org.tynamo.security.jpa.testapp.entities.MyData;
032import org.tynamo.security.jpa.testapp.entities.User;
033import org.tynamo.security.jpa.testapp.pages.NoPermissions;
034import org.tynamo.shiro.extension.realm.text.ExtendedPropertiesRealm;
035
036/**
037 * This module is automatically included as part of the Tapestry IoC Registry, it's a good place to configure and extend Tapestry, or to
038 * place your own service definitions.
039 */
040@SubModule(JpaSecurityModule.class)
041public class AppModule {
042
043        public static void bind(ServiceBinder binder) {
044        }
045
046        public static void contributeApplicationDefaults(MappedConfiguration<String, String> configuration) {
047                configuration.add(SymbolConstants.PRODUCTION_MODE, "false");
048                configuration.add(SymbolConstants.APPLICATION_VERSION, "0.0.1-SNAPSHOT");
049        }
050
051        public static void contributeWebSecurityManager(Configuration<Realm> configuration) {
052                ExtendedPropertiesRealm realm = new ExtendedPropertiesRealm("classpath:users.properties");
053                configuration.add(realm);
054        }
055
056        public static void contributeSeedEntity(OrderedConfiguration<Object> configuration) {
057                User user = new User();
058                user.setId("user");
059                configuration.add("user", user);
060
061                AdminOnly adminOnly = new AdminOnly();
062                configuration.add("adminOnly", adminOnly);
063
064                MyData myData = new MyData();
065                myData.setOwner(user);
066                configuration.add("myData", myData);
067        }
068
069        public void contributeExceptionHandler(MappedConfiguration<Class, Object> configuration) {
070                configuration.add(EntitySecurityException.class, NoPermissions.class);
071        }
072
073}

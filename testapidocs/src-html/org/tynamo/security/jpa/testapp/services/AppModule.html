001package org.tynamo.security.jpa.testapp.services;
002
003import java.sql.SQLException;
004
005import org.apache.shiro.realm.Realm;
006import org.apache.tapestry5.SymbolConstants;
007import org.apache.tapestry5.ioc.Configuration;
008import org.apache.tapestry5.ioc.MappedConfiguration;
009import org.apache.tapestry5.ioc.OrderedConfiguration;
010import org.apache.tapestry5.ioc.ServiceBinder;
011import org.apache.tapestry5.ioc.annotations.Startup;
012import org.apache.tapestry5.ioc.annotations.SubModule;
013import org.apache.tapestry5.ioc.annotations.Symbol;
014import org.tynamo.security.jpa.EntitySecurityException;
015import org.tynamo.security.jpa.JpaSecurityModule;
016import org.tynamo.security.jpa.testapp.entities.AdminOnly;
017import org.tynamo.security.jpa.testapp.entities.MyData;
018import org.tynamo.security.jpa.testapp.entities.User;
019import org.tynamo.security.jpa.testapp.pages.NoPermissions;
020import org.tynamo.shiro.extension.realm.text.ExtendedPropertiesRealm;
021
022/**
023 * This module is automatically included as part of the Tapestry IoC Registry, it's a good place to configure and extend Tapestry, or to
024 * place your own service definitions.
025 */
026@SubModule(JpaSecurityModule.class)
027public class AppModule {
028
029        public static void bind(ServiceBinder binder) {
030        }
031
032        public static void contributeApplicationDefaults(MappedConfiguration<String, String> configuration) {
033                configuration.add(SymbolConstants.PRODUCTION_MODE, "false");
034                configuration.add(SymbolConstants.APPLICATION_VERSION, "0.0.1-SNAPSHOT");
035        }
036
037        public static void contributeWebSecurityManager(Configuration<Realm> configuration) {
038                ExtendedPropertiesRealm realm = new ExtendedPropertiesRealm("classpath:users.properties");
039                configuration.add(realm);
040        }
041
042        public static void contributeSeedEntity(OrderedConfiguration<Object> configuration) {
043                User user = new User();
044                user.setId("user");
045                configuration.add("user", user);
046
047                AdminOnly adminOnly = new AdminOnly();
048                configuration.add("adminOnly", adminOnly);
049
050                MyData myData = new MyData();
051                myData.setOwner(user);
052                configuration.add("myData", myData);
053        }
054
055        public void contributeRequestExceptionHandler(MappedConfiguration<Class, Object> configuration) {
056                configuration.add(EntitySecurityException.class, NoPermissions.class);
057        }
058
059        @Startup
060        public static void startH2WebServer(@Symbol(SymbolConstants.PRODUCTION_MODE) boolean productionMode)
061                throws SQLException {
062                if (!productionMode)
063                        org.h2.tools.Server.createWebServer(new String[] { "-web", "-webAllowOthers", "-webPort", "8082" }).start();
064        }
065
066}

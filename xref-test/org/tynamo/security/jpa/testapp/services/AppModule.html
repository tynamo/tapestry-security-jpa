22  /**
23   * This module is automatically included as part of the Tapestry IoC Registry, it's a good place to configure and extend Tapestry, or to
24   * place your own service definitions.
25   */
26  @SubModule(JpaSecurityModule.class)
27  public class AppModule {
28  
29  	public static void bind(ServiceBinder binder) {
30  	}
31  
32  	public static void contributeApplicationDefaults(MappedConfiguration<String, String> configuration) {
33  		configuration.add(SymbolConstants.PRODUCTION_MODE, "false");
34  		configuration.add(SymbolConstants.APPLICATION_VERSION, "0.0.1-SNAPSHOT");
35  	}
36  
37  	public static void contributeWebSecurityManager(Configuration<Realm> configuration) {
38  		ExtendedPropertiesRealm realm = new ExtendedPropertiesRealm("classpath:users.properties");
39  		configuration.add(realm);
40  	}
41  
42  	public static void contributeSeedEntity(OrderedConfiguration<Object> configuration) {
43  		User user = new User();
44  		user.setId("user");
45  		configuration.add("user", user);
46  
47  		AdminOnly adminOnly = new AdminOnly();
48  		configuration.add("adminOnly", adminOnly);
49  
50  		MyData myData = new MyData();
51  		myData.setOwner(user);
52  		configuration.add("myData", myData);
53  	}
54  
55  	public void contributeRequestExceptionHandler(MappedConfiguration<Class, Object> configuration) {
56  		configuration.add(EntitySecurityException.class, NoPermissions.class);
57  	}
58  
59  	@Startup
60  	public static void startH2WebServer(@Symbol(SymbolConstants.PRODUCTION_MODE) boolean productionMode)
61  		throws SQLException {
62  		if (!productionMode)
63  			org.h2.tools.Server.createWebServer(new String[] { "-web", "-webAllowOthers", "-webPort", "8082" }).start();
64  	}

66  }

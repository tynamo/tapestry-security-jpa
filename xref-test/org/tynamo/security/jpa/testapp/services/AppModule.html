19  package org.tynamo.security.jpa.testapp.services;
20  
21  import org.apache.shiro.realm.Realm;
22  import org.apache.tapestry5.SymbolConstants;
23  import org.apache.tapestry5.ioc.Configuration;
24  import org.apache.tapestry5.ioc.MappedConfiguration;
25  import org.apache.tapestry5.ioc.OrderedConfiguration;
26  import org.apache.tapestry5.ioc.ServiceBinder;
27  import org.apache.tapestry5.ioc.annotations.SubModule;
28  import org.tynamo.security.jpa.EntitySecurityException;
29  import org.tynamo.security.jpa.JpaSecurityModule;
30  import org.tynamo.security.jpa.testapp.entities.AdminOnly;
31  import org.tynamo.security.jpa.testapp.entities.MyData;
32  import org.tynamo.security.jpa.testapp.entities.User;
33  import org.tynamo.security.jpa.testapp.pages.NoPermissions;
34  import org.tynamo.shiro.extension.realm.text.ExtendedPropertiesRealm;
35  
36  /**
37   * This module is automatically included as part of the Tapestry IoC Registry, it's a good place to configure and extend Tapestry, or to
38   * place your own service definitions.
39   */
40  @SubModule(JpaSecurityModule.class)
41  public class AppModule {
42  
43  	public static void bind(ServiceBinder binder) {
44  	}
45  
46  	public static void contributeApplicationDefaults(MappedConfiguration<String, String> configuration) {
47  		configuration.add(SymbolConstants.PRODUCTION_MODE, "false");
48  		configuration.add(SymbolConstants.APPLICATION_VERSION, "0.0.1-SNAPSHOT");
49  	}
50  
51  	public static void contributeWebSecurityManager(Configuration<Realm> configuration) {
52  		ExtendedPropertiesRealm realm = new ExtendedPropertiesRealm("classpath:users.properties");
53  		configuration.add(realm);
54  	}
55  
56  	public static void contributeSeedEntity(OrderedConfiguration<Object> configuration) {
57  		User user = new User();
58  		user.setId("user");
59  		configuration.add("user", user);
60  
61  		AdminOnly adminOnly = new AdminOnly();
62  		configuration.add("adminOnly", adminOnly);
63  
64  		MyData myData = new MyData();
65  		myData.setOwner(user);
66  		configuration.add("myData", myData);
67  	}
68  
69  	public void contributeExceptionHandler(MappedConfiguration<Class, Object> configuration) {
70  		configuration.add(EntitySecurityException.class, NoPermissions.class);
71  	}
72  
73  }

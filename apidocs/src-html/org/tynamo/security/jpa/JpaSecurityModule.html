043        @SuppressWarnings("unchecked")
044        @Advise(serviceInterface = EntityManagerSource.class)
045        public static void secureEntityManager(final MethodAdviceReceiver receiver,
046                @Autobuild SecureEntityManagerSourceAdvice advice) throws SecurityException, NoSuchMethodException {
047                receiver.adviseMethod(receiver.getInterface().getMethod("create", new Class[] { String.class }), advice);
048        }
049
050        public static void contributeFactoryDefaults(MappedConfiguration<String, String> configuration) {

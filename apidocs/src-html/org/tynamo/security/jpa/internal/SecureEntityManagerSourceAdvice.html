001package org.tynamo.security.jpa.internal;
002
003import javax.persistence.EntityManager;
004import javax.servlet.http.HttpServletRequest;
005
006import org.apache.tapestry5.ioc.annotations.Inject;
007import org.apache.tapestry5.ioc.annotations.Symbol;
008import org.apache.tapestry5.ioc.services.PropertyAccess;
009import org.apache.tapestry5.plastic.MethodAdvice;
010import org.apache.tapestry5.plastic.MethodInvocation;
011import org.tynamo.security.jpa.JpaSecuritySymbols;
012import org.tynamo.security.services.SecurityService;
013
014public class SecureEntityManagerSourceAdvice implements MethodAdvice {
015        private final SecurityService securityService;
016        private final PropertyAccess propertyAccess;
017        private final HttpServletRequest request;
018        private String realmName;
019        private Class principalType;
020
021        public SecureEntityManagerSourceAdvice(SecurityService securityService, PropertyAccess propertyAccess,
022                HttpServletRequest request, @Inject @Symbol(JpaSecuritySymbols.ASSOCIATED_REALM) String realmName,
023                @Inject @Symbol(JpaSecuritySymbols.ASSOCIATED_PRINCIPALTYPE) String principalType) throws ClassNotFoundException {
024                this.securityService = securityService;
025                this.propertyAccess = propertyAccess;
026                this.request = request;
027                this.realmName = realmName;
028                this.principalType = principalType.isEmpty() ? null : Class.forName(principalType);
029        }
030
031        @Override
032        public void advise(MethodInvocation invocation) {
033                invocation.proceed();
034                EntityManager entityManager = (EntityManager) invocation.getReturnValue();
035                if (entityManager == null) return;
036                invocation.setReturnValue(new SecureEntityManager(securityService, propertyAccess, request, entityManager,
037                        realmName, principalType));
038        }
039
040}

224        private Object getConfiguredPrincipal() {
225                if (!realmName.isEmpty()) {
226                        Collection principals = securityService.getSubject().getPrincipals().fromRealm(realmName);
227                        if (principalType == null) principals.iterator().next();
228                        else {
229                                for (Object availablePrincipal : principals)
230                                        if (availablePrincipal.getClass().isAssignableFrom(principalType)) { return availablePrincipal; }
231                        }
232                }
233                if (principalType != null) {
234                        Object principal = securityService.getSubject().getPrincipals().oneByType(principalType);
235                        if (principal == null)
236                                throw new NullPointerException("Subject is required to have a configured principal of type '" + principalType
237                                        + "' for secure entity relation checks");
238                        return principal;
239                }
240                return securityService.getSubject().getPrincipals().getPrimaryPrincipal();
241        }

243        @SuppressWarnings({ "unchecked", "rawtypes" })
244        private <T> T secureFind(Class<T> entityClass, Object entityId, LockModeType lockMode, Map<String, Object> properties) {
245                String requiredRoleValue = RequiresAnnotationUtil.getRequiredRole(entityClass, Operation.READ);
246
247                if (requiredRoleValue != null && request.isUserInRole(requiredRoleValue))
248                        return delegate.find(entityClass, entityId, lockMode, properties);
249
250                String requiredAssociationValue = RequiresAnnotationUtil.getRequiredAssociation(entityClass, Operation.READ);
251
252                if (requiredAssociationValue == null) {
253                        // proceed as normal if there's neither RequiresRole nor RequiresAssociation, directly return null if role didn't match
254                        if (requiredRoleValue != null) return null;
255                        if (entityId != null) return delegate.find(entityClass, entityId, lockMode, properties);
256                        // even if assocation is not required for read, we can still use it to find the entity
257                        RequiresAssociation annotation = entityClass.getAnnotation(RequiresAssociation.class);
258                        if (annotation == null) return null;
259                        requiredAssociationValue = annotation.value();
260                }
261
262                // return immediately if user is guest
263                if (request.getRemoteUser() == null) return null;
264
265                CriteriaBuilder builder = delegate.getCriteriaBuilder();
266                CriteriaQuery<Object> criteriaQuery = builder.createQuery();
267                Root<?> from = criteriaQuery.from(entityClass);
268                CriteriaQuery<Object> select = criteriaQuery.select(from);
269                Metamodel metamodel = delegate.getMetamodel();
270
271                EntityType entityType = metamodel.entity(entityClass);
272                Type idType;
273                SingularAttribute idAttr;
274                Predicate predicate2 = null;
275
276                Object principal = getConfiguredPrincipal();
277                // throw IllegalArgumentException below if idType doesn't match with principal's type
278
279                // empty string indicates association to "self"
280                if (requiredAssociationValue.isEmpty()) {
281                        idType = entityType.getIdType();
282                        // entityId may be null when finding entity by association
283                        if (entityId == null) entityId = principal;
284                        else if (!entityId.equals(principal)) return null;
285                } else {
286                        String[] associationAttributes = String.valueOf(requiredAssociationValue).split("\\.");
287                        Path<?> path = null;
288                        // find the type of the top entity while traversing the property path
289                        for (String attributeName : associationAttributes) {
290                                path = path == null ? from.get(attributeName) : path.get(attributeName);
291                                Attribute attribute = entityType.getAttribute(attributeName);
292                                // TODO handle if !attribute.isAssociation()
293                                entityType = metamodel.entity(attribute.getJavaType());
294                        }
295
296                        idType = entityType.getIdType();
297                        idAttr = entityType.getId(idType.getJavaType());
298
299                        // TODO handle subject == null
300                        // TODO allow configuring the principal for it rather than using primary
301                        predicate2 = builder.equal(path.get(idAttr.getName()), principal);
302                }
303
304                Predicate predicate1 = null;
305                if (entityId != null) {
306                        idType = entityType.getIdType();
307                        idAttr = entityType.getId(idType.getJavaType());
308                        predicate1 = builder.equal(from.get(idAttr.getName()), entityId);
309                }
310                criteriaQuery.where(predicate1 == null ? predicate2 : predicate2 == null ? predicate1 : builder.and(predicate1,
311                        predicate2));
312                // getSingleResult throws an exception if no results are found, so get the list instead
313                List results = delegate.createQuery(criteriaQuery).getResultList();
314                if (results.size() > 1)
315                        throw new NonUniqueResultException("More than a single result of type " + entityClass.getName() + " found for "
316                                + (entityId == null ? "association " + requiredAssociationValue : "id " + entityId));
317                return (T) (results.size() <= 0 ? null : results.get(0));
318        }

320        private Annotation getAnnotation(Member member, Class annotationType) {
321                return member instanceof Field ? ((Field) member).getAnnotation(annotationType)
322                        : member instanceof Method ? ((Method) member).getAnnotation(annotationType) : null;
323        }

325        @SuppressWarnings({ "unchecked", "rawtypes" })
326        public void checkWritePermissions(final Object entity, Operation writeOperation) {
327                if (ThreadContext.getSecurityManager() == null) return;
328                String requiredRoleValue = RequiresAnnotationUtil.getRequiredRole(entity.getClass(), writeOperation);
329
330                if (requiredRoleValue != null && request.isUserInRole(requiredRoleValue)) return;
331
332                String requiredAssociationValue = RequiresAnnotationUtil.getRequiredAssociation(entity.getClass(), writeOperation);
333
334                if (requiredAssociationValue == null) {
335                        // proceed as normal if there's neither RequiresRole nor RequiresAssociation, throw an exception if role didn't match
336                        if (requiredRoleValue == null) return;
337                        else throw new EntitySecurityException("Currently executing subject is not permitted to " + writeOperation
338                                + " entities of type " + entity.getClass().getSimpleName());
339                }
340                // throw exception if user is guest
341                if (request.getRemoteUser() == null)
342                        throw new EntitySecurityException("Guest users are not permitted to " + writeOperation + " entities of type "
343                                + entity.getClass().getSimpleName());
344
345                Metamodel metamodel = delegate.getMetamodel();
346                EntityType entityType = metamodel.entity(entity.getClass());
347                // empty association value indicates association to "self"
348                Object associatedObject = entity;
349                if (!requiredAssociationValue.isEmpty()) {
350                        // find the top entity by traversing the property path
351                        String[] associationAttributes = String.valueOf(requiredAssociationValue).split("\\.");
352                        for (String attributeName : associationAttributes) {
353                                Attribute attribute = entityType.getAttribute(attributeName);
354                                // TODO handle if !attribute.isAssociation()
355                                entityType = metamodel.entity(attribute.getJavaType());
356                                associatedObject = propertyAccess.get(associatedObject, attributeName);
357                                if (associatedObject == null)
358                                        throw new EntitySecurityException("Subject for the required association is not set when executing "
359                                                + writeOperation + " on instance '" + entity + "' of type " + entity.getClass().getSimpleName());
360                        }
361                }
362
363                Type idType = entityType.getIdType();
364                SingularAttribute idAttr = entityType.getId(idType.getJavaType());
365
366                // handle INSERT operation to "self" with a generated id as a special allowed case
367                if (associatedObject == entity && getAnnotation(idAttr.getJavaMember(), GeneratedValue.class) != null
368                        && Operation.INSERT.equals(writeOperation) && propertyAccess.get(entity, idAttr.getName()) == null) return;
369                else {
370                        Object principal = getConfiguredPrincipal();
371                        // TODO throw IllegalArgumentException below if idType doesn't match with principal's type
372
373                        if (!propertyAccess.get(associatedObject, idAttr.getName()).equals(principal))
374                                throw new EntitySecurityException("Currently executing subject is not permitted to " + writeOperation
375                                        + " entities of type " + entity.getClass().getSimpleName() + " because the required association didn't exist");
376                }
377        }

224        private Object getConfiguredPrincipal() {
225                if (!realmName.isEmpty()) {
226                        Collection principals = securityService.getSubject().getPrincipals().fromRealm(realmName);
227                        if (principalType == null) principals.iterator().next();
228                        else {
229                                for (Object availablePrincipal : principals)
230                                        if (availablePrincipal.getClass().isAssignableFrom(principalType)) { return availablePrincipal; }
231                        }
232                }
233                if (principalType != null) {
234                        Object principal = securityService.getSubject().getPrincipals().oneByType(principalType);
235                        if (principal == null)
236                                throw new NullPointerException("Subject is required to have a configured principal of type '" + principalType
237                                        + "' for secure entity relation checks");
238                        return principal;
239                }
240                return securityService.getSubject().getPrincipals().getPrimaryPrincipal();
241        }

243        @SuppressWarnings({ "unchecked", "rawtypes" })
244        private <T> T secureFind(Class<T> entityClass,
245                Object entityId,
246                LockModeType lockMode,
247                Map<String, Object> properties) {
248                String requiredAssociationValue = RequiresAnnotationUtil.getRequiredAssociation(entityClass, Operation.READ);
249                // predicate1 is for finding the entity itself in case entityId was given as an argument or deduced
250                EntityType entityType = delegate.getMetamodel().entity(entityClass);
251                Predicate predicate1 = null;
252
253                if (entityId != null) {
254                        Type idType = entityType.getIdType();
255                        SingularAttribute idAttr = entityType.getId(idType.getJavaType());
256                        CriteriaBuilder builder = delegate.getCriteriaBuilder();
257                        CriteriaQuery<T> criteriaQuery = builder.createQuery(entityClass);
258                        Root<T> from = criteriaQuery.from(entityClass);
259                        predicate1 = builder.equal(from.get(idAttr.getName()), entityId);
260                }
261                // getSingleResult throws an exception if no results are found, so get the list instead
262                List results = SecureFinder.find(delegate, request,
263                        SecureFinder.getConfiguredPrincipal(securityService, realmName, principalType),
264                        requiredAssociationValue,
265                        predicate1, entityClass,
266                        entityId, lockMode,
267                        properties);
268                if (results.size() > 1)
269                        throw new NonUniqueResultException("More than a single result of type " + entityClass.getName() + " found for "
270                                + (entityId == null ? "association " + requiredAssociationValue : "id " + entityId));
271                return (T) (results.size() <= 0 ? null : results.get(0));
272        }

274        private Annotation getAnnotation(Member member, Class annotationType) {
275                return member instanceof Field ? ((Field) member).getAnnotation(annotationType)
276                        : member instanceof Method ? ((Method) member).getAnnotation(annotationType) : null;
277        }

279        @SuppressWarnings({ "unchecked", "rawtypes" })
280        public void checkWritePermissions(final Object entity, Operation writeOperation) {
281                if (ThreadContext.getSecurityManager() == null) return;
282                String requiredRoleValue = RequiresAnnotationUtil.getRequiredRole(entity.getClass(), writeOperation);
283
284                if (requiredRoleValue != null && request.isUserInRole(requiredRoleValue)) return;
285
286                String requiredAssociationValue = RequiresAnnotationUtil.getRequiredAssociation(entity.getClass(), writeOperation);
287
288                if (requiredAssociationValue == null) {
289                        // proceed as normal if there's neither RequiresRole nor RequiresAssociation, throw an exception if role didn't match
290                        if (requiredRoleValue == null) return;
291                        else throw new EntitySecurityException("Currently executing subject is not permitted to " + writeOperation
292                                + " entities of type " + entity.getClass().getSimpleName());
293                }
294                // throw exception if user is guest
295                if (request.getRemoteUser() == null)
296                        throw new EntitySecurityException("Guest users are not permitted to " + writeOperation + " entities of type "
297                                + entity.getClass().getSimpleName());
298
299                Metamodel metamodel = delegate.getMetamodel();
300                EntityType entityType = metamodel.entity(entity.getClass());
301                // empty association value indicates association to "self"
302                Object associatedObject = entity;
303                if (!requiredAssociationValue.isEmpty()) {
304                        // find the top entity by traversing the property path
305                        String[] associationAttributes = String.valueOf(requiredAssociationValue).split("\\.");
306                        for (String attributeName : associationAttributes) {
307                                Attribute attribute = entityType.getAttribute(attributeName);
308
309                                if (!attribute.isAssociation() && !attribute.isCollection()) throw new EntityNotFoundException(
310                                        "association " + requiredAssociationValue + " does not exist for base type" + entityType.getName());
311                                // TODO we only support many-to-many relations as the top entity
312                                if (attribute.isCollection()) {
313                                        // for collection attributes, the simplest, although not the most performant, approach
314                                        // is to query the principal entity, then break
315                                        entityType = metamodel.entity(((PluralAttribute) attribute).getBindableJavaType());
316                                        associatedObject = find(entityType.getJavaType(), getConfiguredPrincipal());
317                                        break;
318                                } else {
319                                        entityType = metamodel.entity(attribute.getJavaType());
320                                        associatedObject = propertyAccess.get(associatedObject, attributeName);
321                                }
322                                if (associatedObject == null)
323                                        throw new EntitySecurityException("Subject for the required association is not set when executing "
324                                                + writeOperation + " on instance '" + entity + "' of type " + entity.getClass().getSimpleName());
325                        }
326                }
327
328                Type idType = entityType.getIdType();
329                SingularAttribute idAttr = entityType.getId(idType.getJavaType());
330
331                // handle INSERT operation to "self" with a generated id as a special allowed case
332                if (associatedObject == entity && getAnnotation(idAttr.getJavaMember(), GeneratedValue.class) != null
333                        && Operation.INSERT.equals(writeOperation) && propertyAccess.get(entity, idAttr.getName()) == null) return;
334                else {
335                        Object principal = getConfiguredPrincipal();
336                        // TODO throw IllegalArgumentException below if idType doesn't match with principal's type
337
338                        if (!propertyAccess.get(associatedObject, idAttr.getName()).equals(principal))
339                                throw new EntitySecurityException("Currently executing subject is not permitted to " + writeOperation
340                                        + " entities of type " + entity.getClass().getSimpleName() + " because the required association didn't exist");
341                }
342        }

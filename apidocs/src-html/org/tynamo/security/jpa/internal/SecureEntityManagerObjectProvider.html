034        public <T> T provide(final Class<T> objectType, final AnnotationProvider annotationProvider,
035                final ObjectLocator locator) {
036                if (objectType.equals(EntityManager.class)) return objectType.cast(getOrCreateProxy(annotationProvider, locator));
037
038                return null;
039        }
040
041        private synchronized EntityManager getOrCreateProxy(final AnnotationProvider annotationProvider,
042                final ObjectLocator objectLocator) {
043                if (proxy == null) {
044                        final PlasticProxyFactory proxyFactory = objectLocator.getService("PlasticProxyFactory",
045                                PlasticProxyFactory.class);
046
047                        final PersistenceContext annotation = annotationProvider.getAnnotation(PersistenceContext.class);
048                        final EntityManagerManager entityManagerManager = objectLocator.getService(EntityManagerManager.class);
049                        final SecurityService securityService = objectLocator.getService(SecurityService.class);
050                        final PropertyAccess propertyAccess = objectLocator.getService(PropertyAccess.class);
051                        final HttpServletRequest request = objectLocator.getService(HttpServletRequest.class);
052
053                        proxy = proxyFactory.createProxy(EntityManager.class, new ObjectCreator() {
054                                public Object createObject() {
055                                        final EntityManagerManager entityManagerManager = objectLocator.getService(EntityManagerManager.class);
056
057                                        return new SecureEntityManager(securityService, propertyAccess, request, JpaInternalUtils.getEntityManager(
058                                                entityManagerManager, annotation), "", null);
059                                }
060                        }, "<EntityManagerProxy>");
061                }
062
063                return proxy;
064        }

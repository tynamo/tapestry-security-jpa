031/**
032 * SecureFinder provides common functionality for finding entities securely. We are advising (the object that creates) EntityManager, but we
033 * need to share the implemention with AssociatedEntities service (that provides secure find for all entities of certain types) because
034 * EntityManager API doesn't include any operation for findAll().

039        public static Object getConfiguredPrincipal(SecurityService securityService, String realmName, Class principalType) {
040                if (securityService.getSubject() == null || securityService.getSubject().getPrincipals() == null) return null;
041                if (!realmName.isEmpty()) {
042                        Collection principals = securityService.getSubject().getPrincipals().fromRealm(realmName);
043                        if (principalType == null) principals.iterator().next();
044                        else {
045                                for (Object availablePrincipal : principals)
046                                        if (availablePrincipal.getClass().isAssignableFrom(principalType)) { return availablePrincipal; }
047                        }
048                }
049                if (principalType != null) {
050                        Object principal = securityService.getSubject().getPrincipals().oneByType(principalType);
051                        if (principal == null)
052                                throw new NullPointerException("Subject is required to have a configured principal of type '" + principalType
053                                        + "' for secure entity relation checks");
054                        return principal;
055                }
056                return securityService.getSubject().getPrincipals().getPrimaryPrincipal();
057        }

059        @SuppressWarnings({ "unchecked", "rawtypes" })
060        public static <T> List<T> find(EntityManager delegate,
061                HttpServletRequest request,
062                Object principal,
063                String requiredAssociationValue,
064                Predicate predicate1,
065                Class<T> entityClass,
066                Object entityId,
067                LockModeType lockMode,
068                Map<String, Object> properties) {
069                String requiredRoleValue = RequiresAnnotationUtil.getRequiredRole(entityClass, Operation.READ);
070
071                if (requiredRoleValue != null && request.isUserInRole(requiredRoleValue))
072                        return Stream.of(delegate.find(entityClass, entityId, lockMode, properties)).filter(Objects::nonNull)
073                                .collect(Collectors.toList());
074
075                if (requiredAssociationValue == null) {
076                        // proceed as normal if there's neither RequiresRole nor RequiresAssociation, directly return null if role didn't match
077                        if (requiredRoleValue != null) return Collections.emptyList();
078                        if (entityId != null)
079                                return Stream.of(delegate.find(entityClass, entityId, lockMode, properties)).filter(Objects::nonNull)
080                                        .collect(Collectors.toList());
081                        // even if assocation is not required for read, we can still use it to find the entity
082                        RequiresAssociation annotation = entityClass.getAnnotation(RequiresAssociation.class);
083                        if (annotation == null) return Collections.emptyList();
084                        requiredAssociationValue = annotation.value();
085                }
086
087                // return immediately if user is guest
088                if (request.getRemoteUser() == null) return Collections.emptyList();
089
090                CriteriaBuilder builder = delegate.getCriteriaBuilder();
091                CriteriaQuery<T> criteriaQuery = builder.createQuery(entityClass);
092                Root<T> from = criteriaQuery.from(entityClass);
093                CriteriaQuery<T> select = criteriaQuery.select(from);
094                Metamodel metamodel = delegate.getMetamodel();
095
096                EntityType entityType = metamodel.entity(entityClass);
097                Type idType;
098                SingularAttribute idAttr;
099                Predicate predicate2 = null;
100
101                // Object principal = getConfiguredPrincipal();
102                // throw IllegalArgumentException below if idType doesn't match with principal's type
103
104                // empty string indicates association to "self"
105                if (requiredAssociationValue.isEmpty()) {
106                        idType = entityType.getIdType();
107                        // entityId may be null when finding entity by association
108                        if (entityId == null) entityId = principal;
109                        else if (!entityId.equals(principal)) return Collections.emptyList();
110                        // handle search to self as a special case. The can be at most one 
111                        return Stream.of(delegate.find(entityClass, entityId, lockMode, properties)).filter(Objects::nonNull)
112                                .collect(Collectors.toList());
113                } else {
114                        String[] associationAttributes = String.valueOf(requiredAssociationValue).split("\\.");
115                        Path<?> path = null;
116
117                        // find the type of the top entity while traversing the property path
118                        for (String attributeName : associationAttributes) {
119                                path = path == null ? from.get(attributeName) : path.get(attributeName);
120                                Attribute attribute = entityType.getAttribute(attributeName);
121                                if (!attribute.isAssociation() && !attribute.isCollection()) throw new EntityNotFoundException(
122                                        "association " + requiredAssociationValue + " does not exist for base type" + entityType.getName());
123                                if (attribute.isCollection()) {
124                                        Root<?> pathRoot = criteriaQuery.from(entityType);
125                                        entityType = metamodel.entity(((PluralAttribute) attribute).getBindableJavaType());
126                                        path = pathRoot.join(attributeName);
127                                } else entityType = metamodel.entity(attribute.getJavaType());
128                        }
129
130                        idType = entityType.getIdType();
131                        idAttr = entityType.getId(idType.getJavaType());
132
133                        // TODO handle subject == null
134                        predicate2 = builder.equal(path.get(idAttr.getName()), principal);
135
136                }
137
138                // predicate1 is for finding the entity itself in case entityId was given as an argument or deduced
139                // Predicate predicate1 = null;
140                // if (entityId != null) {
141                // idType = entityType.getIdType();
142                // idAttr = entityType.getId(idType.getJavaType());
143                // predicate1 = builder.equal(from.get(idAttr.getName()), entityId);
144                // }
145                criteriaQuery
146                        .where(predicate1 == null ? predicate2 : predicate2 == null ? predicate1 : builder.and(predicate1, predicate2));
147                // getSingleResult throws an exception if no results are found, so get the list instead
148                // List results = delegate.createQuery(criteriaQuery).getResultList();
149                return delegate.createQuery(criteriaQuery).getResultList();
150                // if (results.size() > 1)
151                // throw new NonUniqueResultException("More than a single result of type " + entityClass.getName() + " found for "
152                // + (entityId == null ? "association " + requiredAssociationValue : "id " + entityId));
153                // return (T) (results.size() <= 0 ? null : results.get(0));
154        }

001package org.tynamo.security.jpa.internal;
002
003import javax.persistence.EntityManager;
004import javax.persistence.metamodel.Attribute;
005import javax.persistence.metamodel.EntityType;
006import javax.persistence.metamodel.Metamodel;
007import javax.persistence.metamodel.SingularAttribute;
008import javax.persistence.metamodel.Type;
009import javax.servlet.http.HttpServletRequest;
010
011import org.apache.tapestry5.ioc.services.PropertyAccess;
012import org.apache.tapestry5.plastic.MethodAdvice;
013import org.apache.tapestry5.plastic.MethodInvocation;
014import org.tynamo.security.jpa.EntitySecurityException;
015import org.tynamo.security.jpa.annotations.Operation;
016import org.tynamo.security.services.SecurityService;
017
018@Deprecated
019public class SecureWriteAdvice implements MethodAdvice {
020        private final SecurityService securityService;
021        private final HttpServletRequest request;
022        private final PropertyAccess propertyAccess;
023        private final Operation writeOperation;
024
025        public SecureWriteAdvice(final Operation writeOperation, final SecurityService securityService,
026                final HttpServletRequest request, final PropertyAccess propertyAccess) {
027                if (!Operation.INSERT.equals(writeOperation) && !Operation.UPDATE.equals(writeOperation)
028                        && !Operation.DELETE.equals(writeOperation))
029                        throw new IllegalArgumentException("The targeted writeOperation must be one of INSERT, UPDATE, DELETE but was "
030                                + writeOperation);
031                this.writeOperation = writeOperation;
032                this.securityService = securityService;
033                this.request = request;
034                this.propertyAccess = propertyAccess;
035        }
036
037        @SuppressWarnings({ "unchecked", "rawtypes" })
038        @Override
039        public void advise(MethodInvocation invocation) {
040                Object entity = invocation.getParameter(0);
041                String requiredRoleValue = RequiresAnnotationUtil.getRequiredRole(entity.getClass(), writeOperation);
042
043                if (requiredRoleValue != null && request.isUserInRole(requiredRoleValue)) {
044                        invocation.proceed();
045                        return;
046                }
047
048                String requiredAssociationValue = RequiresAnnotationUtil.getRequiredAssociation(entity.getClass(), writeOperation);
049
050                if (requiredAssociationValue == null) {
051                        // proceed as normal if there's neither RequiresRole nor RequiresAssociation, throw an exception if role didn't match
052                        if (requiredRoleValue == null) invocation.proceed();
053                        else throw new EntitySecurityException("Currently executing subject is not permitted to " + writeOperation
054                                + " entities of type " + entity.getClass().getSimpleName());
055                }
056                EntityManager entityManager = (EntityManager) invocation.getInstance();
057
058                // FIXME handle empty value, i.e. association to "self"
059                Object associatedObject = propertyAccess.get(entity, requiredAssociationValue);
060
061                Metamodel metamodel = entityManager.getMetamodel();
062                EntityType entityType = metamodel.entity(entity.getClass());
063                Attribute association = entityType.getAttribute(requiredAssociationValue);
064                entityType = metamodel.entity(association.getJavaType());
065                Type idType = entityType.getIdType();
066                SingularAttribute idAttr = entityType.getId(idType.getJavaType());
067                if (!propertyAccess.get(associatedObject, idAttr.getName()).equals(
068                        securityService.getSubject().getPrincipals().getPrimaryPrincipal())) { throw new EntitySecurityException(
069                        "Currently executing subject is not permitted to " + writeOperation + " entities of type "
070                                + entity.getClass().getSimpleName() + " because the required association didn't exist"); }
071                invocation.proceed();
072        }
073}

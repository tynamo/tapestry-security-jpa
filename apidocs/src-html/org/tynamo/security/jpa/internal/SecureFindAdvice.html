001package org.tynamo.security.jpa.internal;
002
003import javax.persistence.EntityManager;
004import javax.persistence.criteria.CriteriaBuilder;
005import javax.persistence.criteria.CriteriaQuery;
006import javax.persistence.criteria.Path;
007import javax.persistence.criteria.Predicate;
008import javax.persistence.criteria.Root;
009import javax.persistence.metamodel.Attribute;
010import javax.persistence.metamodel.EntityType;
011import javax.persistence.metamodel.Metamodel;
012import javax.persistence.metamodel.SingularAttribute;
013import javax.persistence.metamodel.Type;
014import javax.servlet.http.HttpServletRequest;
015
016import org.apache.tapestry5.plastic.MethodAdvice;
017import org.apache.tapestry5.plastic.MethodInvocation;
018import org.tynamo.security.jpa.annotations.Operation;
019import org.tynamo.security.jpa.annotations.RequiresRole;
020import org.tynamo.security.services.SecurityService;
021
022@Deprecated
023public class SecureFindAdvice implements MethodAdvice {
024        private SecurityService securityService;
025        private HttpServletRequest request;
026
027        public SecureFindAdvice(final SecurityService securityService, HttpServletRequest request) {
028                this.securityService = securityService;
029                this.request = request;
030        }
031
032        @SuppressWarnings({ "unchecked", "rawtypes" })
033        @Override
034        public void advise(MethodInvocation invocation) {
035                Class aClass = (Class) invocation.getParameter(0);
036                RequiresRole requiresRole = (RequiresRole) aClass.getAnnotation(RequiresRole.class);
037                String requiredRoleValue = RequiresAnnotationUtil.getRequiredRole(aClass, Operation.READ);
038
039                if (requiredRoleValue != null && request.isUserInRole(requiredRoleValue)) {
040                        invocation.proceed();
041                        return;
042                }
043
044                String requiredAssociationValue = RequiresAnnotationUtil.getRequiredAssociation(aClass, Operation.READ);
045
046                if (requiredAssociationValue == null) {
047                        // proceed as normal if there's neither RequiresRole nor RequiresAssociation, directly return null if role didn't match
048                        if (requiredRoleValue == null) invocation.proceed();
049                        else invocation.setReturnValue(null);
050                        return;
051                }
052                EntityManager entityManager = (EntityManager) invocation.getInstance();
053                // FIXME handle empty value, i.e. association to "self"
054                CriteriaBuilder builder = entityManager.getCriteriaBuilder();
055                CriteriaQuery<Object> criteriaQuery = builder.createQuery();
056                Root<?> from = criteriaQuery.from(aClass);
057                CriteriaQuery<Object> select = criteriaQuery.select(from);
058                Metamodel metamodel = entityManager.getMetamodel();
059
060                EntityType entityType = metamodel.entity(aClass);
061                Type idType;
062                SingularAttribute idAttr;
063
064                Object entityId = invocation.getParameter(1);
065                Predicate predicate1 = null;
066                if (entityId != null) {
067                        idType = entityType.getIdType();
068                        idAttr = entityType.getId(idType.getJavaType());
069                        predicate1 = builder.equal(from.get(idAttr.getName()), invocation.getParameter(1));
070                }
071
072                String[] associationAttributes = String.valueOf(requiredAssociationValue).split("\\.");
073                Path<?> path = null;
074                // find the type of the top entity while traversing the property path
075                for (String attributeName : associationAttributes) {
076                        path = path == null ? from.get(attributeName) : path.get(attributeName);
077                        Attribute attribute = entityType.getAttribute(attributeName);
078                        // TODO handle if !attribute.isAssociation()
079                        entityType = metamodel.entity(attribute.getJavaType());
080                }
081
082                idType = entityType.getIdType();
083                idAttr = entityType.getId(idType.getJavaType());
084
085                // TODO handle subject == null
086                // TODO allow configuring the principal for it rather than using primary
087                Predicate predicate2 = builder.equal(path.get(idAttr.getName()), securityService.getSubject().getPrincipals()
088                        .getPrimaryPrincipal());
089                criteriaQuery.where(predicate1 == null ? predicate2 : builder.and(predicate1, predicate2));
090                invocation.setReturnValue(entityManager.createQuery(criteriaQuery).getSingleResult());
091        }
092}

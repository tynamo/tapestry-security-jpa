001package org.tynamo.security.jpa.services;
002
003import java.util.List;
004import java.util.concurrent.Callable;
005
006import javax.persistence.EntityManager;
007import javax.servlet.http.HttpServletRequest;
008
009import org.apache.tapestry5.ioc.annotations.Inject;
010import org.apache.tapestry5.ioc.annotations.Symbol;
011import org.tynamo.security.jpa.JpaSecuritySymbols;
012import org.tynamo.security.jpa.annotations.Operation;
013import org.tynamo.security.jpa.internal.RequiresAnnotationUtil;
014import org.tynamo.security.jpa.internal.SecureFinder;
015import org.tynamo.security.services.SecurityService;
016
017public class AssociatedEntitiesImpl implements AssociatedEntities {
018        private final EntityManager delegate;
019        private final SecurityService securityService;
020        private final HttpServletRequest request;
021        private String realmName;
022        private Class<?> principalType;
023
024        public AssociatedEntitiesImpl(
025                final SecurityService securityService,
026                final HttpServletRequest request, final EntityManager delegate,
027                @Inject @Symbol(JpaSecuritySymbols.ASSOCIATED_REALM) String realmName,
028                @Inject @Symbol(JpaSecuritySymbols.ASSOCIATED_PRINCIPALTYPE) String principalType) throws ClassNotFoundException {
029                this.securityService = securityService;
030                this.request = request;
031                this.delegate = delegate;
032                this.realmName = realmName;
033                this.principalType = principalType.isEmpty() ? null : Class.forName(principalType);
034        }
035
036        @Override
037        public <T> List<T> findAll(EntityManager em, Class<T> entityClass) throws Exception {
038                String requiredAssociationValue = RequiresAnnotationUtil.getRequiredAssociation(entityClass, Operation.READ);
039
040                // return SecureFinder.find(delegate, request, SecureFinder.getConfiguredPrincipal(securityService,
041                // realmName,
042                // principalType),
043                // requiredAssociationValue, null, entityClass,
044                // null, null, null);
045                
046                Object principal = SecureFinder.getConfiguredPrincipal(securityService, realmName, principalType);
047                // invoke with security disabled so we don't go to a recursion with em.find()
048                return securityService.invokeWithSecurityDisabled(new Callable<List<T>>() {
049                        @Override
050                        public List<T> call() {
051                                return SecureFinder.find(delegate, request, principal,
052                                        requiredAssociationValue,
053                                        null, entityClass, null, null, null);
054                        }
055                });
056                
057        }
058
059}
